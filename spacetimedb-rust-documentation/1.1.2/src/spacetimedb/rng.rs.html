25    /// Retrieve the random number generator for this reducer transaction,
26    /// seeded by the timestamp of the reducer call.
27    ///
28    /// If you only need a single random value, you can use [`ReducerContext::random()`].
29    ///
30    /// # Examples
31    ///
32    /// ```no_run
33    /// # #[cfg(target_arch = "wasm32")] mod demo {
34    /// use spacetimedb::{reducer, ReducerContext};
35    /// use rand::Rng;
36    ///
37    /// #[spacetimedb::reducer]
38    /// fn rng_demo(ctx: &spacetimedb::ReducerContext) {
39    ///     // Can be used in method chaining style:
40    ///     let digit = ctx.rng().gen_range(0..=9);
41    ///
42    ///     // Or, cache locally for reuse:
43    ///     let mut rng = ctx.rng();
44    ///     let floats: Vec<f32> = rng.sample_iter(rand::distributions::Standard).collect();
45    /// }
46    /// # }
47    /// ```
48    ///
49    /// For more information, see [`StdbRng`] and [`rand::Rng`].
50    pub fn rng(&self) -> &StdbRng {
51        self.rng.get_or_init(|| StdbRng {
52            rng: StdRng::seed_from_u64(self.timestamp.to_micros_since_unix_epoch() as u64).into(),
53            _marker: PhantomData,
54        })
55    }
56}

58/// A reference to the random number generator for this reducer call.
59///
60/// An instance can be obtained via [`ReducerContext::rng()`]. Import
61/// [`rand::Rng`] in order to access many useful random algorithms.
62///
63/// `StdbRng` uses the same PRNG as `rand`'s [`StdRng`]. Note, however, that
64/// because it is seeded from a publically-known timestamp, it is not
65/// cryptographically secure.
66///
67/// You may be looking for a level of reproducibility that's finer-grained
68/// than "if it happens at the exact same time, you get the same result" --
69/// if so, just seed an [`StdRng`] directly, or use another rng like those
70/// listed [here](https://rust-random.github.io/book/guide-rngs.html).
71/// Just note that you must not store any state, including an rng, in a global
72/// variable or any other in-WASM side channel. Any and all state persisted
73/// across reducer calls _must_ be stored in the database.
74pub struct StdbRng {
75    // Comments in the rand crate claim RefCell can have an overhead of up to 15%,
76    // and so they use an UnsafeCell instead:
77    // <https://docs.rs/rand/0.8.5/src/rand/rngs/thread.rs.html#20-32>
78    // This is safe as long as no method on `StdRngCell` is re-entrant.
79    rng: UnsafeCell<StdRng>,
80
81    // !Send + !Sync
82    _marker: PhantomData<*mut ()>,
83}
